module glowny_sterownik_silnika(
	input clk,
	input wire switch_2,
	input wire switch_6,
	input wire switch_5,
	output wire LED7,
	output wire [7:0] seven_segmented_display,
	output wire [2:0] zalaczony_wyswietlacz
    );
	
	parameter integer taktowanie = 12000000;
	reg [6:0] rpm;
	reg [6:0] we_rpm;
	reg start = 0;
	wire [6:0] wy_rpm_1;
	wire [6:0] wy_rpm_2;
	wire [25:0] taktowanie_na_stopien;
	wire sygnal_dodania_rpm;
	wire sygnal_odjecia_rpm;
	wire rozruch;
	
	rozrusznik rozrusznik(.clk(clk), .switch_6(switch_6), .taktowanie(taktowanie), .switch_2(switch_2), .rozruch(rozruch));
	
	dodanie_5_rpm dodatnie_5_rpm(.switch_6(switch_6), .clk(clk), .we_rpm(we_rpm), .rpm(wy_rpm_1), .sygnal_zmiany_rpm(sygnal_dodania_rpm));
	
	odjecie_5_rpm odjecie_5_rpm(.switch_5(switch_5), .clk(clk), .we_rpm(we_rpm), .rpm(wy_rpm_2), .sygnal_zmiany_rpm(sygnal_odjecia_rpm));
	
	//obliczanie_taktowania_na_stopien oblicz_taktowanie_na_stopien(.clk(clk), .rpm(rpm), .taktowanie_na_stopien(taktowanie_na_stopien));
	
	dzielenie dzielenie(.clk(clk), .B(rpm), .wynik(taktowanie_na_stopien));
	
	wyswietlanie_rpm_na_wyswietlaczach wyswietlanie_rpm(.clk(clk), .rpm(rpm), .seven_segmented_display(seven_segmented_display), .zalaczony_wyswietlacz(zalaczony_wyswietlacz));
	
	cyl_1_zawor_wydechowy cyl_1_zawor_wydechowy(.clk(clk), .rpm(rpm), .sygnal_dodania_rpm(sygnal_dodania_rpm), .sygnal_odjecia_rpm(sygnal_odjecia_rpm), .rozruch(rozruch), .taktowanie_na_stopien(taktowanie_na_stopien), .LED7(LED7));
	
	always @(posedge clk) begin
	
		if (sygnal_dodania_rpm == 1) begin
			rpm = wy_rpm_1;
		end
		else if (sygnal_odjecia_rpm == 1) begin
			rpm = wy_rpm_2;
		end
		
		if (sygnal_dodania_rpm == 0 && sygnal_odjecia_rpm == 0 && start == 1) begin
			we_rpm = rpm;
		end
		
		if (rozruch == 1)
			start = 1;
		
		if (start == 0) begin
			rpm = 0;
			we_rpm = 0;
		end
	end
endmodule
